    defop _dist x1 as floating, x2 as floating, 
			y1 as floating, y2 as floating returns floating
        sqrt (+ (pow (- x2 x1) 2) (y2 -> - y1 -> pow 2)) -> Out
     
    defcom Init
		Let xp be integer
		Let yp be integer
        41 -> Assign xp
        42 -> Assign yp
        _dist (_getX) (_getY) xp yp -> Assign dist
        dist -> Print
    
    defcom Loop
        Ahead(100)
        TurnGunRight(180)
        Ahead(100)
        TurnGunRight(180)
end
